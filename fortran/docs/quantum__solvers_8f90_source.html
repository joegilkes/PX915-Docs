284  real(real64), dimension(3) :: r_temp
285  real(real64) :: c_temp
286  real(real64), dimension(2, 3) :: r_diffs ! Wavefunction finite differences
287  real(real64), dimension(2) :: c_diffs
288  real(real64) :: wfn, wfnx2 ! Wavefunction at walker position
289  integer :: i
290 
291  ! Calculate trial wavefunction at provided point.
292  wfn = h_2_plus_wfn(c, r, r_a, r_b)
293  wfnx2 = 2 * wfn
294 
295  ! Calculate trial wavefunction in finite difference directions.
296  do i = 1, 3
297  r_temp(:) = r(:)
298  r_temp(i) = r(i) - h2plus%ham_fdstep
299  r_diffs(1, i) = h_2_plus_wfn(c, r_temp, r_a, r_b)
300 
301  r_temp(i) = r(i) + h2plus%ham_fdstep
302  r_diffs(2, i) = h_2_plus_wfn(c, r_temp, r_a, r_b)
303  end do
304 
305  ! Assemble Hamiltonian
306  laplacian = ((r_diffs(1, 1) - wfnx2 + r_diffs(2, 1)) + &
307  (r_diffs(1, 2) - wfnx2 + r_diffs(2, 2)) + &
308  (r_diffs(1, 3) - wfnx2 + r_diffs(2, 3))) / (h2plus%ham_fdstep**2)
309  laplacian = laplacian * (-0.5_real64)
310 
311  coulomb = -(1 / norm2(r - r_a))*wfn - (1 / norm2(r - r_b))*wfn
312 
313  e_loc = ((laplacian + coulomb) / wfn) + (1 / norm2(r_a - r_b))
314 
315  ! Calculate wavefunction gradient wrt c
316  c_temp = c - h2plus%c_fdstep
317  c_diffs(1) = h_2_plus_wfn(c_temp, r, r_a, r_b)
318 
319  c_temp = c + h2plus%c_fdstep
320  c_diffs(2) = h_2_plus_wfn(c_temp, r, r_a, r_b)
321 
322  gradient = (c_diffs(2) - c_diffs(1)) / (2.0_real64 * h2plus%c_fdstep)
323 
324  end subroutine h_2_plus_energy

347  ! Calculates a value for a given the bond length s by Newton-Raphson iteration
348  ! over the Lambert W function.
349  function h_2_cusp(s) result(a)
350 
351  implicit none
352 
353  real(real64), intent(in) :: s
354  real(real64) :: a
355 
356  real(real64) :: f, f_dash, emsoa
357  integer :: maxiter = 1000
358  real(real64) :: tol = 1e-5_real64
359  integer :: k
360 
361  a = 0.5_real64 ! guess based on s=0 solution
362 
363  if (write_log) then
364  write(logid, "('Starting Newton-Raphson search for parameter a...')")
365  write(logid, *) 'a abs(f)'
366  end if
367 
368  f = (1.0_real64 / (1.0_real64 + exp(-s/a))) - a
369  do k = 1, maxiter
370 
371  if (abs(f) .lt. tol) then
372  if (write_log) then
373  write(logid, "('Value of parameter a converged.')")
374  write(logid, "('')")
375  end if
376  exit
377  end if
378 
379  emsoa = exp(-s/a)
380  f = (1.0_real64 / (1.0_real64 + emsoa)) - a
381  f_dash = (-s * emsoa) / ((1.0_real64 + emsoa*emsoa * a*a)) - 1
382 
383  a = a - (f / f_dash)
384 
385  if (write_log) then
386  write(logid, "(F12.8, ' ', F12.8)") a, abs(f)
387  end if
388 
389  end do
390 
391  ! Crash out if not converged
392  if (k .ge. maxiter) then
393  error stop "Newton-Raphson search for parameter a has failed to converge."
394  end if
395 
396  end function h_2_cusp
