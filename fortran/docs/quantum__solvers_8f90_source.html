283  real(real64), dimension(3) :: r_temp
284  real(real64), dimension(2, 3) :: diffs ! Wavefunction finite differences
285  real(real64) :: wfn, wfnx2 ! Wavefunction at walker position
286  integer :: i
287 
288  real(real64) :: e_loc
289 
290  ! Allow the finite difference step size to take a default value.
291  if (present(fdstep)) then
292  step = fdstep
293  else
294  step = 1e-3_real64
295  end if
296 
297  ! Calculate trial wavefunction at provided point.
298  wfn = h_2_plus_wfn(c, r, r_a, r_b)
299  wfnx2 = 2 * wfn
300 
301  ! Calculate trial wavefunction in finite difference directions.
302  do i = 1, 3
303  r_temp(:) = r(:)
304  r_temp(i) = r(i) - step
305  diffs(1, i) = h_2_plus_wfn(c, r_temp, r_a, r_b)
306 
307  r_temp(i) = r(i) + step
308  diffs(2, i) = h_2_plus_wfn(c, r_temp, r_a, r_b)
309  end do
310 
311  ! Assemble Hamiltonian
312  laplacian = ((diffs(1, 1) - wfnx2 + diffs(2, 1)) + &
313  (diffs(1, 2) - wfnx2 + diffs(2, 2)) + &
314  (diffs(1, 3) - wfnx2 + diffs(2, 3))) / (step*step)
315  laplacian = laplacian * (-0.5_real64)
316 
317  coulomb = -(1 / norm2(r - r_a))*wfn - (1 / norm2(r - r_b))*wfn &
318  + (1 / norm2(r_a - r_b))*wfn
319 
320  e_loc = (laplacian + coulomb) / wfn
321 
322  end function h_2_plus_energy
