54  integer(int64), parameter :: ir = 2836, &
55  ndiv = 1 + floor((real(im, kind=real64)-1.0)/ntab)
56  real(real64), parameter :: am = 1.0/real(im, kind=real64), eps = 1.2e-7, &
57  rnmx = 1.0-eps

59  ! Loop counters
60  integer(int64) :: j, k
61 
62  ! The following is transcribed directly from Press et al
63  if (dseed<=0 .or. iy==0) then ! initialization
64  dseed = max(-dseed,1)
65 
66  do j = ntab+8, 1, -1
67  k = floor(real(dseed, kind=real64)/real(iq, kind=real64))
68  dseed = ia * (dseed-k*iq) - ir * k
69  if (dseed < 0) dseed = dseed + im
70  if (j <= ntab) iv(j) = dseed
71  end do
72 
73  iy = iv(1)
74  end if
75 
76  k = floor(real(dseed, kind=real64)/real(iq, kind=real64))
77  dseed = ia * (dseed-k*iq) - ir * k
78  if (dseed < 0) dseed = dseed + im
79  j = 1_int64 + iy / ndiv
80  iy = iv(j)
81  iv(j) = dseed
82  sample = min(am*iy, rnmx)
