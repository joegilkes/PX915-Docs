108  integer(int64), parameter :: ia = 16807, im = 2147483647, iq = 127773
109  integer(int64), parameter :: ir = 2836, &
110  ndiv = 1 + floor((real(im, kind=real64)-1.0)/ntab)
111  real(real64), parameter :: am = 1.0/real(im, kind=real64), eps = 1.2e-7, &
112  rnmx = 1.0-eps
113 
114  ! Loop counters
115  integer(int64) :: j, k
116 
117  ! The following is transcribed directly from Press et al
118  if (dseed<=0 .or. iy==0) then ! initialization
119  ! NB this may need to be modified to allow for restarting as iy will be 0
120  dseed = max(-dseed,1)
121 
122  do j = ntab+8, 1, -1
123  k = floor(real(dseed, kind=real64)/real(iq, kind=real64))
124  dseed = ia * (dseed-k*iq) - ir * k
125  if (dseed < 0) dseed = dseed + im
126  if (j <= ntab) iv(j) = dseed
127  end do
128 
129  iy = iv(1)
130  end if
131 
132  k = floor(real(dseed, kind=real64)/real(iq, kind=real64))
133  dseed = ia * (dseed-k*iq) - ir * k
134  if (dseed < 0) dseed = dseed + im
135  j = 1_int64 + iy / ndiv
136  iy = iv(j)
137  iv(j) = dseed
138  sample = min(am*iy, rnmx)
139 
140  end subroutine ran1

266  real(real64) :: unif_num ! A random sample from the uniform distribution
267  real(real64), dimension(1) :: x_start ! The starting point of the MCMC chain
268  real(real64), dimension(1) :: X_n ! The proposed step
269  real(real64), dimension(1) :: sample_normal ! A sample from the normal distribution
270  integer :: accept ! Number of accepted steps
271  integer :: i ! Loop variable
272  character(len=1) :: acceptance ! Yes/No signpost for acceptance
273 
274  allocate(x(1:n+1)); x(:) = 0.0_real64
275  allocate(energy_chain(1:n+1)); energy_chain(:) = 0.0_real64
276 
277  ! Creates the initial position of the random walker
278  call random_normal(1, sample_normal)
279  x_start = sample_normal
280  x(1) = x_start(1)
281 
282  ! Calculates the local energy of the initial position
283  energy_chain(1) = qho_energy(alpha_const, x(1))
284 
285  ! Start the counter for the number of accepted moves
286  accept = 0
287 
288  if (verbose) then
289  write(logid, "('Markov chain for alpha = ', F7.4)") alpha_const
290  write(logid, "('Position Local Energy Accepted?')")
291  write(logid, "(F12.8, ' ', F12.8)") x(1), energy_chain(1)
292  end if
293 
294  ! Metropolis Algorithm
295  do i = 2, n+1
296  ! Propose the next step
297  call random_normal(1, sample_normal)
298  x_n = x(i-1) + sigma * sample_normal
299 
300  ! Calculate the transition probability between proposed step and current step
301  log_prob_n = log(qho_prob(alpha_const, x(i-1), x_n(1)))
302 
303  ! Calculate the acceptance ratio
304  a_prob_trans = min(1.0_real64, exp(log_prob_n))
305 
306  ! Generate sample from uniform distribution
307  call ran1(unif_num, seed)
308 
309  ! Compare the drawn sample from the uniform distribution with the acceptance ratio
310  if (a_prob_trans >= unif_num) then
311  ! If accepted calculate local energy and add proposed position to the MCMC chain
312  x(i) = x_n(1)
313  energy_chain(i) = qho_energy(alpha_const, x_n(1))
314 
315  acceptance = 'Y'
316  ! Increment by the number of accepted steps
317  accept = 1 + accept
318  else
319  ! If rejected the proposed step is the current step
320  x(i) = x(i-1)
321  ! The local energy of the proposed step is the same as the current step when rejected
322  energy_chain(i) = energy_chain(i-1)
323 
324  acceptance = 'N'
325  end if
326 
327  if (verbose) then
328  if (mod(i, 1) .eq. 0.0) then
329  write(logid, "(F12.8, ' ', F12.8, ' ', A1)") &
330  x(i), energy_chain(i), acceptance
331  end if
332  end if
333  end do
334 
335  ! if (verbose) write(logid, "('')")
336  write(*, *) '[Temp output] variable accept = ', accept
337  if (verbose) write(0, "('')")
338 
339  ! Calculate the acceptance ratio
340  accept_rate = accept / n
341 
342  end subroutine vqmc_qho
