86  ! Constants defined in Press et al
87  integer(int64), parameter :: ia = 16807, im = 2147483647, iq = 127773
88  integer(int64), parameter :: ir = 2836, &
89  ndiv = 1 + floor((real(im, kind=real64)-1.0)/ntab)
90  real(real64), parameter :: am = 1.0/real(im, kind=real64), eps = 1.2e-7, &
91  rnmx = 1.0-eps
92 
93  ! Loop counters
94  integer(int64) :: j, k
95 
96  ! The following is transcribed directly from Press et al
97  if (dseed<=0 .or. iy==0) then ! initialization
98  ! NB this may need to be modified to allow for restarting as iy will be 0
99  dseed = max(-dseed,1)
100 
101  do j = ntab+8, 1, -1
102  k = floor(real(dseed, kind=real64)/real(iq, kind=real64))
103  dseed = ia * (dseed-k*iq) - ir * k
104  if (dseed < 0) dseed = dseed + im
105  if (j <= ntab) iv(j) = dseed
106  end do
107 
108  iy = iv(1)
109  end if
110 
111  k = floor(real(dseed, kind=real64)/real(iq, kind=real64))
112  dseed = ia * (dseed-k*iq) - ir * k
113  if (dseed < 0) dseed = dseed + im
114  j = 1_int64 + iy / ndiv
115  iy = iv(j)
116  iv(j) = dseed
117  sample = min(am*iy, rnmx)
