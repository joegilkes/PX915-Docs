179  '''
180  @brief

183  @details

206  ax1.fill_between(alpha_array, energies+stdevs, energies-stdevs, alpha=0.3, color='purple')
207  ax1.axvline(0.5, ls='-', color='black', alpha=0.3)
208  ax1.set_xlabel(r'$\alpha$')
209  ax1.set_ylabel('Energy')
210 
211  fig.show()
212
