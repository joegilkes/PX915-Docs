175 def orb_1s(X,Y,Z, R_a): #R_a is the location of the atoms
176  x_a,y_a,z_a = R_a
177  a0 = 1
178  orbital = 1/(((a0**1.5)*(np.sqrt(np.pi)))*\
179  np.exp(np.sqrt((X-x_a)**2+(Y-y_a)**2+(Z-z_a)**2)/a0))
180  return orbital
