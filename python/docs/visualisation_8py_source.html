188 def orb_1s(X,Y,Z, R_a): #R_a is the location of the atoms
189  x_a,y_a,z_a = R_a
190  a0 = 1
191  orbital = 1/(((a0**1.5)*(np.sqrt(np.pi)))*\
192  np.exp(np.sqrt((X-x_a)**2+(Y-y_a)**2+(Z-z_a)**2)/a0))
193  return orbital

198  return orbital
