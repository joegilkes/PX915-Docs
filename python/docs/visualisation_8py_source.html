224 def orb_1s(X,Y,Z, R_a): #R_a is the location of the atoms
225  x_a,y_a,z_a = R_a
226  a0 = 1
227  orbital = 1/(((a0**1.5)*(np.sqrt(np.pi)))*\
228  np.exp(np.sqrt((X-x_a)**2+(Y-y_a)**2+(Z-z_a)**2)/a0))
229  return orbital

231 def orb_2s(r,R_a):
232  a0 = 1
233  orbital = (2-np.linalg.norm(r-R_a)()/a0)/((4*(a0**1.5)*(np.sqrt(2*np.pi)))*np.exp(np.linalg.norm(r-R_a)/(2*a0)))
234  return orbital
